FIX 1: src/components/DashboardLayout.tsx

1a. The handleSearchSubmit function currently does:
  navigate("/clients");
Change it to:
  navigate(`/clients?q=${encodeURIComponent(searchQuery.trim())}`);

1b. The desktop search <input> has no keyboard submit handler. Add onKeyDown to it:
  onKeyDown={(e) => {
    if (e.key === "Enter" && searchQuery.trim()) {
      navigate(`/clients?q=${encodeURIComponent(searchQuery.trim())}`);
    }
  }}

1c. Add a global keyboard shortcut for ⌘K / Ctrl+K. In the DashboardLayout component body, add a useEffect:
  useEffect(() => {
    const handler = (e: KeyboardEvent) => {
      if ((e.metaKey || e.ctrlKey) && e.key === "k") {
        e.preventDefault();
        // On desktop: focus the search input
        const input = document.querySelector<HTMLInputElement>('input[placeholder="Cari klien, order, trip…"]');
        if (input) input.focus();
        // On mobile: open search overlay
        else setSearchOpen(true);
      }
    };
    window.addEventListener("keydown", handler);
    return () => window.removeEventListener("keydown", handler);
  }, []);

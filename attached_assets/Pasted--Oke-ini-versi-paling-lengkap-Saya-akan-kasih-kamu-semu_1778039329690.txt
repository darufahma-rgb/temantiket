2. Kode Lengkap RouteTimeline.tsx
tsx// src/components/RouteTimeline.tsx
import React from 'react';

interface Stop {
  city: string;
  airport: string;
  time?: string;
  flightNumber?: string;
}

interface RouteTimelineProps {
  outbound: {
    origin: Stop;
    destination: Stop;
  };
  returnTrip?: {
    origin: Stop;
    destination: Stop;
  };
  type: 'oneWay' | 'roundTrip';
}

export const RouteTimeline: React.FC<RouteTimelineProps> = ({
  outbound,
  returnTrip,
  type,
}) => {
  const renderLeg = (leg: { origin: Stop; destination: Stop }, label?: string) => (
    <div className="flex gap-4">
      {/* Timeline Column */}
      <div className="flex flex-col items-center pt-1">
        {/* Origin - Dashed Circle */}
        <div className="w-3.5 h-3.5 rounded-full border-[2.5px] border-dashed border-[#1a56a8] bg-white" />
        
        {/* Vertical Dashed Line */}
        <div className="w-px flex-1 border-l-[2.5px] border-dashed border-gray-300 my-1" />
        
        {/* Plane Icon */}
        <div className="text-[#1a56a8] text-xl rotate-90 my-1">✈︎</div>
        
        {/* Vertical Dashed Line */}
        <div className="w-px flex-1 border-l-[2.5px] border-dashed border-gray-300 my-1" />
        
        {/* Destination - Solid Circle */}
        <div className="w-3.5 h-3.5 rounded-full bg-[#1a56a8]" />
      </div>

      {/* City Info */}
      <div className="flex flex-col justify-between flex-1 text-sm py-0.5">
        <div>
          <div className="font-semibold text-gray-900">{leg.origin.city}</div>
          <div className="text-xs text-gray-500">{leg.origin.airport}</div>
          {leg.origin.time && <div className="text-xs text-gray-600 mt-0.5">{leg.origin.time}</div>}
        </div>

        <div className="mt-auto pt-4">
          <div className="font-semibold text-gray-900">{leg.destination.city}</div>
          <div className="text-xs text-gray-500">{leg.destination.airport}</div>
          {leg.destination.time && <div className="text-xs text-gray-600 mt-0.5">{leg.destination.time}</div>}
        </div>
      </div>
    </div>
  );

  return (
    <div className="space-y-5">
      {/* Outbound */}
      <div>
        {type === 'roundTrip' && (
          <div className="flex items-center gap-2 text-xs font-medium text-gray-500 mb-2">
            <span>↗</span> BERANGKAT
          </div>
        )}
        {renderLeg(outbound)}
      </div>

      {/* Return */}
      {type === 'roundTrip' && returnTrip && (
        <>
          <div className="border-t border-dashed border-gray-300 my-1" />
          <div>
            <div className="flex items-center gap-2 text-xs font-medium text-gray-500 mb-2">
              <span>↩</span> PULANG
            </div>
            {renderLeg(returnTrip)}
          </div>
        </>
      )}
    </div>
  );
};

3. Kode MultiLegTimeline.tsx (Bonus)
tsx// src/components/MultiLegTimeline.tsx
import React from 'react';

interface Leg {
  origin: string;
  destination: string;
  time?: string;
}

interface MultiLegTimelineProps {
  legs: Leg[];
  label?: string;
}

export const MultiLegTimeline: React.FC<MultiLegTimelineProps> = ({ legs, label }) => {
  return (
    <div>
      {label && <div className="text-xs font-medium text-gray-500 mb-2">{label}</div>}
      <div className="space-y-4">
        {legs.map((leg, index) => (
          <div key={index} className="flex gap-3 items-start">
            <div className="flex flex-col items-center pt-1">
              <div className="w-2.5 h-2.5 rounded-full bg-amber-500" />
              {index !== legs.length - 1 && (
                <div className="w-px flex-1 border-l border-dashed border-gray-300 my-1" />
              )}
            </div>
            <div className="text-sm">
              <span className="font-medium">{leg.origin}</span> → <span className="font-medium">{leg.destination}</span>
              {leg.time && <span className="text-xs text-gray-500 ml-2">{leg.time}</span>}
            </div>
          </div>
        ))}
      </div>
    </div>
  );
};
